\begin{exercice}

\begin{multicols}{2}
\begin{enumerate}
\item $\left\{ \begin{matrix}
   7x+3>0  \\
   5x-9\le 0  \\
\end{matrix} \right.$

   -11x+5<0\,\,\,\,\,\,\,\,\,\,\,  \\
   5\left( 1+4x \right)\ge 7+2x  \\
\end{matrix} \right.$
\item $\left\{ \begin{matrix}
   -x-2\le 0\,\,\,\,\,\,\,\,\,\,\,\,\,\,\,  \\
   -2x+\frac{x}{2}+1\ge -\frac{x}{4}  \\

\end{matrix} \right.$
\item $\left\{ \begin{matrix}
   8x-7\ge \frac{15x-9}{2}  \\

\end{matrix} \right.$

   2(4x+1)\ge 5x+8\,\,\,\,\,\,\,\,\,  \\
   2(3x+2)>5(3x-10)  \\
\end{matrix} \right.$
\item $\left\{ \begin{matrix}
   8x+\frac{14x}{5}\ge 66-\frac{12x}{5}  \\
   \frac{1}{6}\left( \frac{7x}{4}+x \right)>x-\frac{13}{2}  \\

\end{enumerate}
\end{multicols}
\end{exercice}
